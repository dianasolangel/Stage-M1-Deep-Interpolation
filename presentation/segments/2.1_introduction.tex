    \textbf{Advection Equation}

    \begin{equation}
        \begin{cases}
        \frac{\partial u}{\partial t} + a \frac{\partial u}{\partial x} = 0 \quad (x,t) \in \Omega \times (0,T) \\
         u(x,t=0) = u_0(x) \quad x \in \delta \Omega \\
        \end{cases}
    \end{equation}
    \\

    Has an analytical solution equal to: 
    $$
    u(t, x)=u_0(x, x-at)
