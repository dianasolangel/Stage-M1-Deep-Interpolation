\begin{frame}{Semi-Lagrangian Scheme}
    
    \textbf{The Semi-Lagrangian scheme is defined as it follows:}
    \centering
    \begin{equation*}
        u\left(t_{n+1}, x_i\right)=\mathcal{I}_h^m\left(u\left(t_n, x\right)\right)\left(x_i-a \Delta t\right)
    \end{equation*}
